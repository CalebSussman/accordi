    treble_events, bass_events, metadata = await parser.parse_musicxml(path)

        clefs = [c.classes[0] for c in part.flatten().getElementsByClass('Clef')[:3]]

            'index': idx,
            'id': getattr(part, 'id', None),
            'partName': getattr(part, 'partName', None),
            'clefs': clefs,
            'measure_count': len(part.getElementsByClass('Measure')),
